/**
 * Guesty Booking Engine API Token Management Example
 *
 * This Node.js script demonstrates how to:
 * - Request an access token using client credentials
 * - Cache the token locally to avoid unnecessary renewals
 * - Reuse the token in subsequent API requests
 * - Refresh the token 5 minutes before it expires
 *
 * Prerequisites:
 * - Node.js installed
 * - Your Guesty API Client ID and Client Secret
 */

const axios = require('axios');
const fs = require('fs');
const path = require('path');

// === Configuration ===
const TOKEN_FILE = path.join(__dirname, 'guesty-token.json'); // Where the token will be cached
const CLIENT_ID = 'YOUR_CLIENT_ID';       // Replace with your Guesty application client ID
const CLIENT_SECRET = 'YOUR_CLIENT_SECRET'; // Replace with your Guesty application client secret
const TOKEN_URL = 'https://booking.guesty.com/oauth2/token';  // Token request endpoint

/**
 * Checks if a valid token is cached. If so, returns it.
 * Otherwise, calls the function to request a new token.
 */
async function getCachedToken() {
  if (fs.existsSync(TOKEN_FILE)) {
    const { access_token, expires_at } = JSON.parse(fs.readFileSync(TOKEN_FILE));
    if (Date.now() < expires_at) {
      return access_token; // Token still valid
    }
  }
  return await requestNewToken(); // Token expired or missing
}

/**
 * Requests a new token from the Guesty API and caches it with an expiration timestamp.
 */
async function requestNewToken() {
  const params = new URLSearchParams();
  params.append('grant_type', 'client_credentials');
  params.append('scope', 'booking_engine:api');
  params.append('client_id', CLIENT_ID);
  params.append('client_secret', CLIENT_SECRET);

  const response = await axios.post(TOKEN_URL, params, {
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' }
  });

  const { access_token, expires_in } = response.data;

  // Set expiration to 5 minutes before actual expiry
  const expires_at = Date.now() + (expires_in - 300) * 1000;

  fs.writeFileSync(TOKEN_FILE, JSON.stringify({ access_token, expires_at }));
  return access_token;
}

/**
 * Demonstrates how to use the access token in an authenticated request to the Booking API.
 */
async function makeAuthorizedRequest() {
  const token = await getCachedToken();

  const response = await axios.get('https://booking-api.guesty.com/v1/search', {
    headers: {
      'Authorization': `Bearer ${token}`,
      'accept': 'application/json; charset=utf-8'
    },
    params: {
      checkIn: '2025-07-01',
      checkOut: '2025-07-05',
      adults: 2
    }
  });

  console.log('Search results:', response.data);
}

makeAuthorizedRequest();
